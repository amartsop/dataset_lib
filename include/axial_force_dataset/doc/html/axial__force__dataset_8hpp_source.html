314 {

316 
317  for(int i = 0; i < m_file_num; i++)
318  {
319  arma::fmat x_y_mat = m_x_y.at(i); arma::fvec x_vec = x_y_mat.col(0);
320  x_sort.at(i, 0) = x_vec.back(); x_sort.at(i, 1) = (float) i;
321  }
322  ArmaExt::sortrows<arma::fmat>(&x_sort, true);
323 
324  float sampling_period = 1.0f / m_sampling_frequency;
325  int index_max = (float) x_sort.at(x_sort.n_rows-1, 1);
326

328  {

331  resampling(&m_x_y.at(index), sampling_period);
332  }
333  resampling(&m_x_y.at(index_max), sampling_period);
334 
335  // Size of measuremets

337 
338  for(int i = 0; i < m_file_num; i++)

343  map_str_to_variable(m_meas_ind_vars[0][i], x_vec);

453 float AxialForceDataset::linear_extrapolation(float tn, arma::fvec t_vec,
454  arma::fvec f_vec)
455 {
456  float tc = arma::as_scalar(t_vec.row(t_vec.n_rows - 1));
457  float to = arma::as_scalar(t_vec.row(t_vec.n_rows - 2));

466 void AxialForceDataset::resampling(arma::fmat *matr, float ts)
467 {
468  arma::fvec t = (*matr).col(0);
469  arma::fvec tu = arma::regspace<arma::fvec>(t.front(), ts, t.back());
470  arma::fmat mat_u(tu.n_rows, (*matr).n_cols); mat_u.zeros();
471  mat_u(0, arma::span::all) = (*matr)(0, arma::span::all);
472 
473  for (int64_t i = 1; i < mat_u.n_rows; i++)
474  {
475  arma::uvec q1 = arma::find(t < tu(i));
476  arma::uvec q2 = arma::find(t > tu(i));
477 
478  // Indices that tu(i) belongs in
479  int64_t index1 = q1.back(); int64_t index2 = q2.front();
480 
481  // Times that tu(i) belongs in
482  float t1 = t(index1); float t2 = t(index2);
483 
484  // Corresponding values for t1 and t2
485  arma::fvec a1 = (*matr)(index1, arma::span(1, matr->n_cols - 1));
486  arma::fvec a2 = (*matr)(index2, arma::span(1, matr->n_cols - 1));
487 
488  // Interpolation
489  arma::fvec a = a2 - ((a2 - a1) / (t2 - t1)) * (t2 - tu(i));
490 
491  mat_u(i, 0) = tu(i);
492  mat_u(i, arma::span(1, mat_u.n_cols - 1)) = a;
493  }
494 
495  (*matr) = mat_u;
496 }

498 arma::fvec AxialForceDataset::central_diff_derivative(arma::fvec t_vec,
499  arma::fvec x_vec)
500 {
501  arma::fvec u_vec(t_vec.n_rows); u_vec.fill(0);
502 
503  for (u_int64_t i = 0; i < t_vec.n_rows; i++)
504  {
505  float to, tn, xo, xn;
506 
507  if (i == 0)
508  {
509  to = 0.0; tn = t_vec(i + 1);
510  xo = 0.0; xn = x_vec(i + 1);
511  }
512  else if (i == t_vec.n_rows - 1)
513  {
514  to = t_vec(i); tn = 0.0;
515  xo = x_vec(i); xn = 0.0;
516  }
517  else
518  {
519  to = t_vec(i - 1); tn = t_vec(i + 1);
520  xo = x_vec(i - 1); xn = x_vec(i + 1);
521  }
522 
523  float step = tn - to;
524  if (step == 0) { step = 10e-6; }
525  u_vec(i) = (xn - xo) / (2 * step);
526  }
527 
528  return u_vec;
529 }
